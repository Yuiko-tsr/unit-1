'''py
def sum_letters(text:str) -> int:
    alphabet = "abcdefghijklmnopqrstuvwxyz"
    sum_total = 0

    for let in text.lower():
        index = -1
        for i in range(len(alphabet)):
            if let == alphabet[i]:
                index = i + 1
                sum_total += index
    return sum_total

case1 = sum_letters(text="Math")
