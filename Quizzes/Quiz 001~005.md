def DNAtranslator(in_protein: str) -> str:
    out_protein = ""
    for protein in in_protein:
        if protein == "A":
            out_protein += "T"
        elif protein == "G":
            out_protein += "C"
        elif protein == "T":
            out_protein += "A"
        elif protein == "C":
            out_protein += "G"
        else:
            out_protein += ""

    return out_protein


input_sequence = "AGBCT"
output_sequence = DNAtranslator(input_sequence)
print(output_sequence)
```

def sum_letters(text:str) -> int:
    alphabet = "abcdefghijklmnopqrstuvwxyz"
    sum_total = 0

    for let in text.lower():
        index = -1
        for i in range(len(alphabet)):
            if let == alphabet[i]:
                index = i + 1
                sum_total += index
    return sum_total

case1 = sum_letters(text="Math")
print(case1)
```
